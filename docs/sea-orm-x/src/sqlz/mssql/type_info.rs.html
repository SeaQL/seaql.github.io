pub use tiberius::ColumnType as MsSqlTypeInfo;

use crate::TypeInfo;

impl TypeInfo for MsSqlTypeInfo {
    fn is_null(&self) -> bool {
        matches!(self, MsSqlTypeInfo::Null)
    }

    fn name(&self) -> &str {
        match self {
            MsSqlTypeInfo::Null => "NULL",
            MsSqlTypeInfo::Bit => "BIT",
            MsSqlTypeInfo::Int1 => "INT1",
            MsSqlTypeInfo::Int2 => "INT2",
            MsSqlTypeInfo::Int4 => "INT4",
            MsSqlTypeInfo::Int8 => "INT8",
            MsSqlTypeInfo::Datetime4 => "DATETIME4",
            MsSqlTypeInfo::Float4 => "FLOAT4",
            MsSqlTypeInfo::Float8 => "FLOAT8",
            MsSqlTypeInfo::Money => "MONEY",
            MsSqlTypeInfo::Datetime => "DATETIME",
            MsSqlTypeInfo::Money4 => "MONEY4",
            MsSqlTypeInfo::Guid => "GUID",
            MsSqlTypeInfo::Intn => "INTN",
            MsSqlTypeInfo::Bitn => "BITN",
            MsSqlTypeInfo::Decimaln => "DECIMALN",
            MsSqlTypeInfo::Numericn => "NUMERICN",
            MsSqlTypeInfo::Floatn => "FLOATN",
            MsSqlTypeInfo::Datetimen => "DATETIMEN",
            MsSqlTypeInfo::Daten => "DATEN",
            MsSqlTypeInfo::Timen => "TIMEN",
            MsSqlTypeInfo::Datetime2 => "DATETIME2",
            MsSqlTypeInfo::DatetimeOffsetn => "DATETIMEOFFSETN",
            MsSqlTypeInfo::BigVarBin => "BIGVARBIN",
            MsSqlTypeInfo::BigVarChar => "BIGVARCHAR",
            MsSqlTypeInfo::BigBinary => "BIGBINARY",
            MsSqlTypeInfo::BigChar => "BIGCHAR",
            MsSqlTypeInfo::NVarchar => "NVARCHAR",
            MsSqlTypeInfo::NChar => "NCHAR",
            MsSqlTypeInfo::Xml => "XML",
            MsSqlTypeInfo::Udt => "UDT",
            MsSqlTypeInfo::Text => "TEXT",
            MsSqlTypeInfo::Image => "IMAGE",
            MsSqlTypeInfo::NText => "NTEXT",
            MsSqlTypeInfo::SSVariant => "SSVARIANT",
        }
    }
}
