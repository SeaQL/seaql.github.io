use nix::sys::socket::setsockopt;
use nix::sys::socket::sockopt::KeepAlive;
use std::net::TcpListener;

let listener = TcpListener::bind("0.0.0.0:0").unwrap();
let fd = listener;
let res = setsockopt(&fd, KeepAlive, &true);
assert!(res.is_ok());
