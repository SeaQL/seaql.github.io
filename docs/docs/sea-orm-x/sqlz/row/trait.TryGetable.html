Trait sqlz::row::TryGetable

pub trait TryGetable<'a, DB: Database>
where
    Self: Sized + 'a,
{
    // Required method
    fn try_get_from<I>(row: &'a DB::Row, index: I) -> Result<Self, Error>
       where I: ColumnIndex<DB::Row>;
}
