Module nix::errno

Safe wrappers around errno functions

§Example

use nix::errno::Errno;

Errno::EIO.set();
assert_eq!(Errno::last(), Errno::EIO);

Errno::clear();
assert_eq!(Errno::last(), Errno::from_raw(0));
